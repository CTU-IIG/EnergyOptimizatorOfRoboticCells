55 template <class C>
56 uintptr_t orderWiseHash(const C& v) {
57  uintptr_t hash = 1ul, i = 1;
58  constexpr uintptr_t R = 3144134277;
59  for (typename C::const_iterator it = v.cbegin(); it != v.cend(); ++it, ++i) {
60  uintptr_t h = cast(*it);
61  hash *= R+2*h*i;
62  hash ^= h;
63  }
64 
65  hash >>= 2;
66 
67  return hash;
68 }
